function check_player_levelup(exp_amount) {
    player_kills = player_kills + exp_amount;
    
    if (has_bloodlust) {
        blood_kills = blood_kills + exp_amount;
        if (blood_kills >= 5) {
            blood_kills = 0;
            if (player_hp < player_max_hp) { player_hp = player_hp + 1; }
        }
    }

    let need = 3 + 2 * player_level;
    if (player_kills >= need) {
        player_level = player_level + 1;
        player_kills = 0;
        if (player_level / 10 === math_floor(player_level / 10)) {
            player_hp = player_max_hp;
        }
        show_levelup();
    }
}

function player() { // 玩家类
    return message => {
        if (message === "create_player") {
            const player_ins = create_sprite('https://raw.githubusercontent.com/StoreManager-Riddle/SourceAcademyProject/refs/heads/main/npc_elf.png');
            return update_scale(player_ins, [player_scale, player_scale]);
        }
        
        if (message === "update_position") { 
            return player => update_position(player, [width / 2, height / 2]);
        }
        
        if (message === "damage") { 
            return amount => {
                player_hp = player_hp - amount;
                player_hit_timer = 10;
            };
        }

        if (message === "update_color") {
            return player_obj => {
                if (player_hit_timer > 0) {
                    player_hit_timer = player_hit_timer - 1;
                    update_color(player_obj, [255, 0, 0, 255]);
                } else {
                    update_color(player_obj, [255, 255, 255, 255]);
                }
            };
        }
        
        if (message === "determine_direction") { 
            return (object, speed_x) => {
                if (speed_x > 0) {
                    player_dir_sign = 1;
                } else {
                    player_dir_sign = -1;
                }
            };
        }

        return 0;
    };
}

// enemy
function enemy(initial_hp, initial_speed, texture_url, behavior, if_reward) {
    let max_hp = initial_hp;
    let hp = 0;
    const speed = initial_speed;
    let sprite = undefined;
    const enemy_scale = 2.5;
    let hit_timer = 0;
    let move_timer = 0;
    let fire_timer = 30;
    let stun_timer = 0; // 静止电击计时器

    return message => {
        if (message === "init") {
            return () => {
                sprite = create_sprite(texture_url);
                update_scale(sprite, [0, 0]);
                update_position(sprite, [-9999, -9999]);
                return sprite;
            };
        }

        if (message === "spawn") {
            return (start_x, start_y) => {
                hp = math_round(max_hp * (player_level + 1) / 7);
                if (hp < 1) { hp = 1; }
                hit_timer = 0;
                stun_timer = 0;
                move_timer = math_floor(math_random() * 100);
                update_rotation(sprite, 0);
                
                if (behavior === 6) {
                    update_color(sprite, [255, 215, 0, 255]);
                } else {
                    update_color(sprite, [255, 255, 255, 255]);
                }
                
                update_scale(sprite, [-enemy_scale, enemy_scale]);
                update_position(sprite, [start_x, start_y]);
            };
        }

        if (message === "is_alive") {
            return () => hp > 0;
        }

        if (message === "stun") {
            return duration => {
                if (hp > 0) {
                    stun_timer = duration;
                    update_color(sprite, [0, 150, 255, 255]); // 变成湛蓝色
                }
            };
        }

        if (message === "damage") {
            return amount => {
                if (hp > 0) {
                    hp = hp - amount;
                    hit_timer = 10;
                    update_color(sprite, [255, 0, 0, 255]);
                    
                    if (hp <= 0) {
                        let rs = math_random();
                        if (rs < 0.3) { play_audio(die_s1); }
                        else if (rs < 0.6) { play_audio(die_s2); }
                        else { play_audio(die_s3); }

                        let die_x = query_position(sprite)[0];
                        let die_y = query_position(sprite)[1];
                        particles()('emit')(die_x, die_y);

                        if (if_reward === 1) {
                            game_rewards()('spawn')(die_x, die_y);
                        }
                        
                        if (behavior === 5) {
                            particles()('bomb_emit')(die_x, die_y);
                            for (let i = 0; i < array_length(enermys); i = i + 1) {
                                if (enermys[i]('is_alive')()) {
                                    let ep = query_position(enermys[i]('get_sprite')());
                                    let edx = die_x - ep[0];
                                    let edy = die_y - ep[1];
                                    if (math_sqrt(edx * edx + edy * edy) < 53) {
                                        enermys[i]('damage')(math_round(max_hp * 2 / 3));
                                    }
                                }
                            }
                            if (math_abs(die_x - width / 2) < 53 && math_abs(die_y - height / 2) < 53 &&
                                (die_x - width / 2) * (die_x - width / 2) + (die_y - height / 2) * (die_y - height / 2) < 2809) {
                                player()('damage')(2);
                            }
                        }

                        check_player_levelup(1);
                        update_scale(sprite, [0, 0]);
                        update_position(sprite, [-9999, -9999]);
                    }
                }
            };
        }

        if (message === "chase_player") {
            return () => {
                if (sprite !== undefined && hp > 0) {
                    // 处理静止电击状态
                    if (stun_timer > 0) {
                        stun_timer = stun_timer - 1;
                        if (stun_timer % 10 === 0) {
                            let pos = query_position(sprite);
                            particles()('heal_emit')(pos[0], pos[1]); // 生成电击火花特效
                        }
                        if (stun_timer <= 0) {
                            if (behavior === 6) {
                                update_color(sprite, [255, 215, 0, 255]);
                            } else {
                                update_color(sprite, [255, 255, 255, 255]);
                            }
                        }
                        return 0; // 处于静止状态，直接拦截移动
                    }

                    if (hit_timer > 0) {
                        hit_timer = hit_timer - 1;
                        if (hit_timer <= 0) {
                            if (behavior === 6) {
                                update_color(sprite, [255, 215, 0, 255]);
                            } else {
                                update_color(sprite, [255, 255, 255, 255]);
                            }
                        }
                    }
                    move_timer = move_timer + 1;
                    let pos = query_position(sprite);
                    let ex = pos[0];
                    let ey = pos[1];
                    let dx = (width / 2) - ex;
                    let dy = (height / 2) - ey;
                    let dist = math_sqrt(dx * dx + dy * dy);

                    if (dist > 0 && dist <= width / 2 * 1.5) {
                        let dir_x = dx / dist;
                        let dir_y = dy / dist;
                        let mx = 0;
                        let my = 0;
                        let cur_sp = speed;
                        let cycle = 0;
                        if (behavior === 1) {
                            mx = dir_x * speed;
                            my = dir_y * speed;
                        } else if (behavior === 2) {
                            let wave = math_sin(move_timer * 0.08) * 1.5;
                            mx = dir_x * speed + (-dir_y * wave);
                            my = dir_y * speed + (dir_x * wave);
                        } else if (behavior === 3) {
                            cycle = move_timer % 100;
                            if (cycle < 20) { cur_sp = speed * 16; }
                            else if (cycle < 85) { cur_sp = speed; }
                            else { cur_sp = 0; }
                            mx = dir_x * cur_sp;
                            my = dir_y * cur_sp;
                        } else if (behavior === 4) {
                            if (dist > 400) { mx = 0; my = 0; }
                            else if (dist > 300) { mx = dir_x * speed; my = dir_y * speed; }
                            else { mx = 0; my = 0; }
                            fire_timer = fire_timer - 1;
                            if (fire_timer <= 0 && dist <= 300) {
                                enemy_bullets()('fire')(ex, ey, width / 2, height / 2);
                                fire_timer = 60;
                            }
                        } else if (behavior === 5) {
                            mx = dir_x * speed;
                            my = dir_y * speed;
                        } else if (behavior === 6) {
                            if (dist < 200) {
                                mx = -dir_x * speed * 1.5;
                                my = -dir_y * speed * 1.5;
                            }
                        } else if (behavior === 7) {
                            mx = dir_x * speed * 0.4;
                            my = dir_y * speed * 0.4;
                            fire_timer = fire_timer - 1;
                            if (fire_timer <= 0) {
                                fire_timer = 300;
                                particles()('heal_emit')(ex, ey);
                                for (let h = 0; h < array_length(enermys); h = h + 1) {
                                    if (enermys[h]('is_alive')()) {
                                        let hp_pos = query_position(enermys[h]('get_sprite')());
                                        let hdx = hp_pos[0] - ex;
                                        let hdy = hp_pos[1] - ey;
                                        if (hdx * hdx + hdy * hdy < 14400) {
                                            enermys[h]('heal')(2);
                                            particles()('heal_emit')(hp_pos[0], hp_pos[1]);
                                        }
                                    }
                                }
                            }
                        }
                        let wobble = 0;
                        let sy_anim = enemy_scale;
                        if (cur_sp > 0) {
                            let freq = (behavior === 3 && cycle < 20) ? 0.26 : 0.2;
                            wobble = math_sin(move_timer * freq) * 0.10;
                            sy_anim = enemy_scale + math_abs(math_sin(move_timer * freq)) * 0.3;
                        }
                        update_rotation(sprite, wobble);
                        let empx = player_map_px + (ex - width / 2);
                        let empy = player_map_py + (ey - height / 2);
                        let nx = ex;
                        let ny = ey;
                        let tx = math_floor(empx / tile_pixel);
                        let ty = math_floor(empy / tile_pixel);
                        if (!is_wall(tile_map, math_floor((empx + mx) / tile_pixel), ty)) {
                            nx = ex + mx;
                        }
                        if (!is_wall(tile_map, tx, math_floor((empy + my) / tile_pixel))) {
                            ny = ey + my;
                        }
                        update_position(sprite, [nx, ny]);
                        if (dx > 0) {
                            update_scale(sprite, [-enemy_scale, sy_anim]);
                        } else {
                            update_scale(sprite, [enemy_scale, sy_anim]);
                        }
                    }
                }
            };
        }

        if (message === "update_map_offset") {
            return (dx, dy) => {
                if (sprite !== undefined && hp > 0) {
                    let pos = query_position(sprite);
                    update_position(sprite, [pos[0] - dx, pos[1] - dy]);
                }
            };
        }

        if (message === "get_sprite") {
            return () => sprite;
        }

        if (message === "get_behavior") {
            return () => behavior;
        }

        if (message === "heal") {
            return amount => { 
                if (hp > 0) { 
                    hp = hp + amount; 
                    if (hp > max_hp) { hp = max_hp; } 
                } 
            };
        }

        if (message === "push") {
            return (dx, dy) => {
                if (sprite !== undefined && hp > 0) {
                    let pos = query_position(sprite);
                    let nx = pos[0] + dx;
                    let ny = pos[1] + dy;
                    let mpx = player_map_px + (nx - width / 2);
                    let mpy = player_map_py + (ny - height / 2);
                    let tl = math_floor((mpx - 20) / tile_pixel);
                    let tr = math_floor((mpx + 19) / tile_pixel);
                    let tt = math_floor((mpy - 20) / tile_pixel);
                    let tb = math_floor((mpy + 19) / tile_pixel);
                    if (!is_wall(tile_map, tl, tt) && !is_wall(tile_map, tr, tt) &&
                        !is_wall(tile_map, tl, tb) && !is_wall(tile_map, tr, tb)) {
                        update_position(sprite, [nx, ny]);
                    }
                }
            };
        }

        if (message === "set_hp") {
            return target_hp => {
                if (hp > 0) {
                    hp = target_hp;
                }
            };
        }
        return 0;
    };
}

// spawner
function enemy_spawner(spawn_amount, spawn_rate) {
    let spawn_cooldown = spawn_rate;

    return message => {
        if (message === "update") {
            return () => {
                spawn_cooldown = spawn_cooldown - 1;
                if (spawn_cooldown <= 0) {
                    let mult = 1;
                    if (player_level <= 6) {
                        for (let k = 1; k < player_level; k = k + 1) { mult = mult * 1.2; }
                    } else {
                        for (let k = 1; k <= 5; k = k + 1) { mult = mult * 1.2; }
                    }
                    let amount = math_round(spawn_amount * mult);
                    if (player_level > 6) { amount = amount + (player_level - 6); }
                    if (player_level <= 2) { amount = amount + 6; }
                    let spawned = 0;
                    for (let i = 0; i < array_length(enermys); i = i + 1) {
                        if (spawned < amount) {
                            if (!enermys[i]('is_alive')()) {
                                let tries = 0;
                                let ok = false;
                                let sx = 0;
                                let sy = 0;
                                while (!ok && tries < 10) {
                                    let a = math_random() * 2 * math_PI;
                                    let r = 600;
                                    sx = width / 2 + r * math_cos(a);
                                    sy = height / 2 + r * math_sin(a);
                                    let mpx = player_map_px + (sx - width / 2);
                                    let mpy = player_map_py + (sy - height / 2);
                                    let tx = math_floor(mpx / tile_pixel);
                                    let ty = math_floor(mpy / tile_pixel);
                                    if (!is_wall(tile_map, tx, ty)) { ok = true; }
                                    tries = tries + 1;
                                }
                                if (ok) {
                                    enermys[i]('spawn')(sx, sy);
                                    spawned = spawned + 1;
                                }
                            }
                        }
                    }
                    spawn_cooldown = spawn_rate;
                }
            };
        }
        return 0;
    };
}

// reward
function reward(texture_url) {
    let active = false;
    let sprite = undefined;
    let type = 0; // 0: 恢复HP, 1: 获得经验

    return message => {
        if (message === "init") {
            return () => {
                sprite = create_sprite(texture_url);
                update_scale(sprite, [0, 0]);
                update_position(sprite, [-9999, -9999]);
                return sprite;
            };
        }

        if (message === "spawn") {
            return (x, y) => {
                active = true;
                type = math_random() < 0.3 ? 0 : 1; 
                
                update_position(sprite, [x, y]);
                update_scale(sprite, [2.0, 2.0]);
                
                if (type === 0) {
                    update_color(sprite, [255, 0, 0, 255]);
                } else {
                    update_color(sprite, [0, 200, 255, 255]);
                }
            };
        }

        if (message === "update") {
            return () => {
                if (active && sprite !== undefined) {
                    let pos = query_position(sprite);
                    let dx = (width / 2) - pos[0];
                    let dy = (height / 2) - pos[1];
                    let dist_sq = dx * dx + dy * dy;

                    if (dist_sq < 1225) { // 35*35范围内自动吸
                        active = false;
                        update_scale(sprite, [0, 0]);
                        update_position(sprite, [-9999, -9999]);
                        
                        if (type === 0) {
                            player_hp = player_hp + 1;
                            if (player_hp > player_max_hp) { 
                                player_hp = player_max_hp; 
                            }
                        } else {
                            check_player_levelup(12);
                        }
                    }
                }
            };
        }

        if (message === "update_map_offset") {
            return (dx, dy) => {
                if (active && sprite !== undefined) {
                    let pos = query_position(sprite);
                    update_position(sprite, [pos[0] - dx, pos[1] - dy]);
                }
            };
        }

        if (message === "is_active") {
            return () => active;
        }

        return 0;
    };
}

const reward_pool_size = 30;
const global_rewards = [];
const reward_texture = "https://raw.githubusercontent.com/StoreManager-Riddle/SourceAcademyProject/refs/heads/TEST/HPreward.png";

function game_rewards() {
    return message => {
        if (message === "spawn") {
            return (x, y) => {
                for (let i = 0; i < reward_pool_size; i = i + 1) {
                    if (!global_rewards[i]("is_active")()) {
                        global_rewards[i]("spawn")(x, y);
                        break;
                    }
                }
            };
        }
        if (message === "update") {
            return () => {
                for (let i = 0; i < reward_pool_size; i = i + 1) {
                    global_rewards[i]("update")();
                }
            };
        }
        if (message === "update_map_offset") {
            return (dx, dy) => {
                for (let i = 0; i < reward_pool_size; i = i + 1) {
                    global_rewards[i]("update_map_offset")(dx, dy);
                }
            };
        }
        return 0;
    };
}

// map
const floor_texture_density = 0.3;
function generateTileMap(width, height) { 
    const border = 8;
    let map = [];
    
    for (let y = 0; y < height; y = y + 1) {
        map[y] = [];
        for (let x = 0; x < width; x = x + 1) {
            if (y < border || y >= height - border ||
                x < border || x >= width - border) {
                map[y][x] = 'wallTop';
            } else if (y === border) {
                if (x === border) {
                    map[y][x] = 'wallLeftBottom';
                } else if (x === width - border - 1) {
                    map[y][x] = 'bottomRightWall';
                } else {
                    map[y][x] = 'bottomWall';
                }
            } else if (y === height - border - 1) {
                if (x === border) {
                    map[y][x] = 'topLeftWall';
                } else if (x === width - border - 1) {
                    map[y][x] = 'topRightWall';
                } else {
                    map[y][x] = 'topWall';
                }
            } else if (x === border) {
                map[y][x] = 'rightWall';
            } else if (x === width - border - 1) {
                map[y][x] = 'LeftWall';
            } else if (y === border + 1 || y === border + 2) {
                map[y][x] = 'wallCenter';
            } else {
                let r = math_random();
                if (r < floor_texture_density / 2) {
                    map[y][x] = 'texture1Floor';
                } else if (r < floor_texture_density) {
                    map[y][x] = 'texture2Floor';
                } else {
                    map[y][x] = 'floor';
                }
            }
        }
    }
    return map;
}

// ==========================================
// 武器集束炸弹类 (短射程 + 三散射连发)
// ==========================================
function bomb(shrapnel_count, explosion_damage, projectile_texture, weapon_texture) {
    const bomb_pool_size = 5;
    let sprites = [];
    let actives = [];
    let vxs = [];
    let vys = [];
    let timers = [];
    const max_timer = 22;

    return message => {
        if (message === "init") {
            return () => {
                for (let i = 0; i < bomb_pool_size; i = i + 1) {
                    sprites[i] = create_sprite(projectile_texture);
                    update_scale(sprites[i], [0, 0]);
                    update_position(sprites[i], [-9999, -9999]);
                    actives[i] = false;
                    vxs[i] = 0;
                    vys[i] = 0;
                    timers[i] = 0;
                }

                bomb_gun_sprite = create_sprite(weapon_texture);
                update_scale(bomb_gun_sprite, [0, 0]);
                update_position(bomb_gun_sprite, [-9999, -9999]);

                return sprites[0];
            };
        }

        if (message === "fire") {
            return (start_x, start_y, base_angle) => {
                const bomb_count = 3;
                const spread_angle = 0.25;
                let fired = 0;

                for (let i = 0; i < bomb_pool_size && fired < bomb_count; i = i + 1) {
                    if (!actives[i]) {
                        let offset = (fired - 1) * (spread_angle / 2);
                        let a = base_angle + offset;

                        actives[i] = true;
                        timers[i] = max_timer;

                        let bomb_speed = 7;
                        vxs[i] = bomb_speed * math_cos(a);
                        vys[i] = bomb_speed * math_sin(a);

                        update_position(sprites[i], [start_x, start_y]);
                        update_scale(sprites[i], [1.3, 1.3]);
                        update_rotation(sprites[i], a);

                        fired = fired + 1;
                    }
                }
                return 0;
            };
        }

        if (message === "explode") {
            return index => {
                if (!actives[index]) { return 0; }

                let pos = query_position(sprites[index]);
                let bx = pos[0];
                let by = pos[1];

                particles()('bomb_emit')(bx, by);

                let ec = array_length(enermys);
                for (let i = 0; i < ec; i = i + 1) {
                    if (enermys[i]('is_alive')()) {
                        let ep = query_position(enermys[i]('get_sprite')());
                        let edx = bx - ep[0];
                        let edy = by - ep[1];
                        if (math_sqrt(edx * edx + edy * edy) < 70) {
                            enermys[i]('damage')(explosion_damage);
                        }
                    }
                }

                for (let k = 0; k < shrapnel_count; k = k + 1) {
                    let a = (2 * math_PI / shrapnel_count) * k;
                    let bs_idx = -1;
                    for (let b = 0; b < bullet_pool_size; b = b + 1) {
                        if (!b_active[b]) { bs_idx = b; break; }
                    }
                    if (bs_idx !== -1) {
                        b_active[bs_idx] = true;
                        b_type[bs_idx] = 0;
                        update_position(b_sprites[bs_idx], [bx, by]);
                        let bs = bullet_radius / 4;
                        update_scale(b_sprites[bs_idx], [bs, bs]);
                        b_vx[bs_idx] = bullet_speed * math_cos(a);
                        b_vy[bs_idx] = bullet_speed * math_sin(a);
                        b_life[bs_idx] = 0;
                    }
                }

                actives[index] = false;
                update_scale(sprites[index], [0, 0]);
                return update_position(sprites[index], [-9999, -9999]);
            };
        }

        if (message === "update") {
            return () => {
                for (let i = 0; i < bomb_pool_size; i = i + 1) {
                    if (actives[i]) {
                        let pos = query_position(sprites[i]);
                        let nx = pos[0] + vxs[i];
                        let ny = pos[1] + vys[i];
                        let mpx = player_map_px + (nx - width / 2);
                        let mpy = player_map_py + (ny - height / 2);
                        let tx = math_floor(mpx / tile_pixel);
                        let ty = math_floor(mpy / tile_pixel);

                        if (is_wall(tile_map, tx, ty)) {
                            this_bomb('explode')(i);
                        } else {
                            update_position(sprites[i], [nx, ny]);
                            timers[i] = timers[i] - 1;
                            if (timers[i] <= 0) {
                                this_bomb('explode')(i);
                            }
                        }
                    }
                }
                return 0;
            };
        }

        if (message === "update_map_offset") {
            return (dx, dy) => {
                for (let i = 0; i < bomb_pool_size; i = i + 1) {
                    if (actives[i]) {
                        let pos = query_position(sprites[i]);
                        update_position(sprites[i], [pos[0] - dx, pos[1] - dy]);
                    }
                }
                return 0;
            };
        }

        return 0;
    };
}

// 升级UI（完整原本逻辑：随机抽取 3 个不重复的升级选项）
// ==========================================
function show_levelup() {
    let picked = [];
    let pool_size = array_length(upgrade_names);

    for (let i = 0; i < 3; i = i + 1) {
        let found = false;
        let r = 0;
        while (!found) {
            r = math_floor(math_random() * pool_size);
            found = true;
            
            // 排除规则：
            if (bullet_fire_rate <= 2 && r === 2) { found = false; } // 射速已达上限
            if (has_backshot && r === 6) { found = false; }          // 已拥有后射
            if (has_knockback && r === 7) { found = false; }         // 已拥有击退
            if (has_bloodlust && r === 8) { found = false; }         // 已拥有嗜血
            
            // 防止三张卡牌重复
            for (let j = 0; j < array_length(picked); j = j + 1) {
                if (picked[j] === r) { found = false; }
            }
        }
        picked[array_length(picked)] = r;
    }

    // 绘制标题
    update_position(lv_title_card, [width / 2, height / 2 - 80]);
    update_scale(lv_title_card, [3, 0.8]);
    update_color(lv_title_card, [80, 80, 80, 255]);
    update_to_top(lv_title_card);
    update_text(lv_title_text, 'LEVEL UP!');
    update_position(lv_title_text, [width / 2, height / 2 - 80]);
    update_scale(lv_title_text, [1, 1]);
    update_to_top(lv_title_text);

    // 绘制 3 张随机卡牌
    for (let i = 0; i < 3; i = i + 1) {
        let cx = width / 2 + (i - 1) * 180;
        let cy = height / 2;
        update_position(lv_cards[i], [cx, cy]);
        update_scale(lv_cards[i], [3.2, 1.2]);
        update_color(lv_cards[i], [50, 50, 50, 255]);
        update_to_top(lv_cards[i]);

        let opt_name = upgrade_names[picked[i]];
        update_text(lv_texts[i], opt_name);
        
        // 稀有/特殊武器卡牌使用金色字体高亮
        if (opt_name === 'Multishot -15%dmg' || opt_name === 'FireRate' || 
            opt_name === 'PrepareForBoth' || opt_name === 'KeepAway' || 
            opt_name === 'ClusterBomb' || opt_name === 'GatlingGun' || 
            opt_name === 'Revolver' || opt_name === 'TaserGun') {
            update_color(lv_texts[i], [255, 200, 0, 255]);
        } else {
            update_color(lv_texts[i], [255, 255, 255, 255]);
        }

        update_position(lv_texts[i], [cx, cy]);
        update_scale(lv_texts[i], [1, 1]);
        update_to_top(lv_texts[i]);

        levelup_options[i] = picked[i];
    }

    levelup_active = true;
    levelup_click_lock = true;
}

// 修复后的真实应用属性升级函数
function apply_upgrade(index) {
    let opt = upgrade_names[index];
    if (opt === 'Multishot -15%dmg') { bullets_per_shot = bullets_per_shot + 2; bullet_damage = bullet_damage * 0.85; bullet_total_spread = bullet_total_spread + 0.15; }
    if (opt === 'Range')            { bullet_range = bullet_range + 120; bullet_life = math_round(bullet_range / bullet_speed); }
    if (opt === 'FireRate')         { bullet_fire_rate = bullet_fire_rate - 8; if (bullet_fire_rate < 2) { bullet_fire_rate = 2; } }
    if (opt === 'Size +40%dmg')     { bullet_radius = bullet_radius + 2; bullet_damage = bullet_damage * 1.4; }
    if (opt === 'Damage +50%')      { bullet_damage = bullet_damage * 1.5; }
    if (opt === 'Move Speed')       { player_speed = player_speed + 1.5; }
    if (opt === 'PrepareForBoth')   { has_backshot = true; }
    if (opt === 'KeepAway')         { has_knockback = true; }
    if (opt === 'Bloodlust')        { has_bloodlust = true; }
    
    if (opt === 'ClusterBomb')      { is_gatling_mode = false; is_revolver_mode = false; is_taser_mode = false; is_bomb_mode = true; bomb_ammo = 10; }
    if (opt === 'GatlingGun')        { is_bomb_mode = false; is_revolver_mode = false; is_taser_mode = false; is_gatling_mode = true; gatling_ammo = 200; }
    if (opt === 'Revolver')          { is_bomb_mode = false; is_gatling_mode = false; is_taser_mode = false; is_revolver_mode = true; revolver_ammo = 10; }
    if (opt === 'TaserGun')          { is_bomb_mode = false; is_gatling_mode = false; is_revolver_mode = false; is_taser_mode = true; taser_ammo = 3; }
}

// enemy bullet
function enemy_bullets() {
    return message => {
        if (message === "fire") {
            return (fx, fy, tx, ty) => {
                let dx = tx - fx;
                let dy = ty - fy;
                let dist = math_sqrt(dx * dx + dy * dy);
                if (dist > 0) {
                    let done = false;
                    for (let i = 0; i < eb_pool_size && !done; i = i + 1) {
                        if (!eb_active[i]) {
                            update_position(eb_sprites[i], [fx, fy]);
                            update_color(eb_sprites[i], [255, 100, 100, 255]);
                            update_scale(eb_sprites[i], [1, 1]);
                            eb_active[i] = true;
                            eb_vx[i] = (dx / dist) * eb_speed;
                            eb_vy[i] = (dy / dist) * eb_speed;
                            eb_life[i] = 0;
                            done = true;
                        }
                    }
                }
            };
        }

        if (message === "update_all") {
            return () => {
                for (let i = 0; i < eb_pool_size; i = i + 1) {
                    if (eb_active[i]) {
                        eb_life[i] = eb_life[i] + 1;
                        let pos = query_position(eb_sprites[i]);
                        let nx = pos[0] + eb_vx[i];
                        let ny = pos[1] + eb_vy[i];
                        update_position(eb_sprites[i], [nx, ny]);
                        let pdx = nx - width / 2;
                        let pdy = ny - height / 2;
                        let hit = false;
                        if (math_abs(pdx) < 15 && math_abs(pdy) < 15 &&
                            pdx * pdx + pdy * pdy < 225) {
                            hit = true;
                            if (player_hurt_cd <= 0) {
                                player()('damage')(1);
                                player_hurt_cd = player_hurt_interval;
                            }
                        }
                        let mpx = player_map_px + (nx - width / 2);
                        let mpy = player_map_py + (ny - height / 2);
                        let tx = math_floor(mpx / tile_pixel);
                        let ty = math_floor(mpy / tile_pixel);
                        if (eb_life[i] >= eb_max_life || is_wall(tile_map, tx, ty) || hit) {
                            eb_active[i] = false;
                            update_scale(eb_sprites[i], [0, 0]);
                            update_position(eb_sprites[i], [-9999, -9999]);
                        }
                    }
                }
            };
        }

        if (message === "update_map_offset") {
            return (dx, dy) => {
                for (let i = 0; i < eb_pool_size; i = i + 1) {
                    if (eb_active[i]) {
                        let pos = query_position(eb_sprites[i]);
                        update_position(eb_sprites[i], [pos[0] - dx, pos[1] - dy]);
                    }
                }
            };
        }
        return 0;
    };
}

function particles() {
    return message => {
        if (message === "bomb_emit") {
            return (x, y) => {
                let emitted = 0;
                for (let i = 0; i < particle_pool_size && emitted < 24; i = i + 1) {
                    if (!p_active[i]) {
                        let a = math_random() * 2 * math_PI;
                        let sp = 4 * math_random();
                        update_position(p_sprites[i], [x, y]);
                        update_color(p_sprites[i], [255, 100, 20, 255]);
                        update_scale(p_sprites[i], [1.5, 1.5]);
                        p_active[i] = true;
                        p_vx[i] = sp * math_cos(a);
                        p_vy[i] = sp * math_sin(a);
                        p_life[i] = 30;
                        p_is_bomb[i] = true; p_is_heal[i] = false;
                        emitted = emitted + 1;
                    }
                }
            };
        }

        if (message === "emit") {
            return (x, y) => {
                let emitted = 0;
                for (let i = 0; i < particle_pool_size && emitted < particle_count; i = i + 1) {
                    if (!p_active[i]) {
                        let a = math_random() * 2 * math_PI;
                        let sp = particle_speed * math_random();
                        update_position(p_sprites[i], [x, y]);
                        update_color(p_sprites[i], [255, 255, 255, 255]);
                        update_scale(p_sprites[i], [1, 1]);
                        p_active[i] = true;
                        p_vx[i] = sp * math_cos(a);
                        p_vy[i] = sp * math_sin(a);
                        p_life[i] = particle_life;
                        p_is_bomb[i] = false; p_is_heal[i] = false;
                        emitted = emitted + 1;
                    }
                }
            };
        }

        if (message === "heal_emit") {
            return (x, y) => {
                let emitted = 0;
                for (let i = 0; i < particle_pool_size && emitted < 12; i = i + 1) {
                    if (!p_active[i]) {
                        let a = math_random() * 2 * math_PI;
                        let sp = 2 * math_random();
                        update_position(p_sprites[i], [x, y]);
                        update_color(p_sprites[i], [0, 255, 0, 255]);
                        update_scale(p_sprites[i], [2.5, 2.5]);
                        p_active[i] = true; p_vx[i] = sp * math_cos(a); p_vy[i] = sp * math_sin(a);
                        p_life[i] = 25; p_is_bomb[i] = false; p_is_heal[i] = true;
                        emitted = emitted + 1;
                    }
                }
            };
        }

        if (message === "update_all") {
            return () => {
                for (let i = 0; i < particle_pool_size; i = i + 1) {
                    if (p_active[i]) {
                        let pos = query_position(p_sprites[i]);
                        let nx = pos[0] + p_vx[i];
                        let ny = pos[1] + p_vy[i];
                        update_position(p_sprites[i], [nx, ny]);
                        p_life[i] = p_life[i] - 1;
                        let max_life = particle_life;
                        if (p_is_bomb[i]) { max_life = 30; }
                        let a = math_round(255 * p_life[i] / max_life);
                        if (p_is_bomb[i]) {
                            update_color(p_sprites[i], [255, 100, 20, a]);
                        } else if (p_is_heal[i]) {
                            update_color(p_sprites[i], [50, 255, 50, a]);
                        } else {
                            update_color(p_sprites[i], [255, 255, 255, a]);
                        }
                        if (p_life[i] <= 0) {
                            p_active[i] = false;
                            update_position(p_sprites[i], [-9999, -9999]);
                            update_scale(p_sprites[i], [0, 0]);
                        }
                    }
                }
            };
        }
        return 0;
    };
}

function gun() {
    const gun_offset = 20;
    
    return message => {
        if (message === "create_gun") {
            normal_gun_sprite = create_sprite('https://raw.githubusercontent.com/StoreManager-Riddle/SourceAcademyProject/refs/heads/TEST/AK47.png');
            return update_scale(normal_gun_sprite, [gun_scale, gun_scale]);
        }
        
        if (message === "update_gun") {
            return gun_unused => {
                let mouse_pos = query_pointer_position();
                let dx = mouse_pos[0] - width / 2;
                let dy = mouse_pos[1] - height / 2;
                let angle = math_atan2(dy, dx);
                let gun_x = width / 2 + gun_offset * math_cos(angle);
                let gun_y = height / 2 + gun_offset * math_sin(angle);

                // 先将所有武器隐形
                update_scale(normal_gun_sprite, [0, 0]);
                if (bomb_gun_sprite !== undefined) { update_scale(bomb_gun_sprite, [0, 0]); }
                if (gatling_gun_sprite !== undefined) { update_scale(gatling_gun_sprite, [0, 0]); }
                if (revolver_gun_sprite !== undefined) { update_scale(revolver_gun_sprite, [0, 0]); }
                if (taser_gun_sprite !== undefined) { update_scale(taser_gun_sprite, [0, 0]); }

                // 根据模式显隐对应的武器
                if (is_bomb_mode && bomb_gun_sprite !== undefined) {
                    update_scale(bomb_gun_sprite, [gun_scale * 2.5, gun_scale * 2.5]);
                    update_position(bomb_gun_sprite, [gun_x, gun_y]);
                    update_rotation(bomb_gun_sprite, angle);
                } else if (is_gatling_mode && gatling_gun_sprite !== undefined) {
                    update_scale(gatling_gun_sprite, [gun_scale * 1.5, gun_scale * 1.5]);
                    update_position(gatling_gun_sprite, [gun_x, gun_y]);
                    update_rotation(gatling_gun_sprite, angle);
                } else if (is_revolver_mode && revolver_gun_sprite !== undefined) {
                    update_scale(revolver_gun_sprite, [gun_scale * 1.2, gun_scale * 1.2]);
                    update_position(revolver_gun_sprite, [gun_x, gun_y]);
                    update_rotation(revolver_gun_sprite, angle);
                } else if (is_taser_mode && taser_gun_sprite !== undefined) {
                    update_scale(taser_gun_sprite, [gun_scale * 1.3, gun_scale * 1.3]);
                    update_position(taser_gun_sprite, [gun_x, gun_y]);
                    update_rotation(taser_gun_sprite, angle);
                } else {
                    update_scale(normal_gun_sprite, [gun_scale, gun_scale]);
                    update_position(normal_gun_sprite, [gun_x, gun_y]);
                    update_rotation(normal_gun_sprite, angle);
                }
                return 0;
            };
        }
        return 0;
    };
}

function bullets() {
    const spawn_dist = 30;

    return message => {
        if (message === "fire") {
            return mouse_pos => {
                let dx = mouse_pos[0] - width / 2;
                let dy = mouse_pos[1] - height / 2;
                let base = math_atan2(dy, dx);
                let step = bullets_per_shot > 1
                    ? bullet_total_spread / (bullets_per_shot - 1)
                    : 0;
                let half = bullet_total_spread / 2;
                let fired = 0;
                for (let i = 0; i < bullet_pool_size && fired < bullets_per_shot; i = i + 1) {
                    if (!b_active[i]) {
                        let a = base - half + fired * step;
                        let sx = width / 2 + spawn_dist * math_cos(a);
                        let sy = height / 2 + spawn_dist * math_sin(a);
                        let bs = bullet_radius / 4;
                        update_scale(b_sprites[i], [bs, bs]);
                        update_rotation(b_sprites[i], a + math_PI / 2);
                        update_color(b_sprites[i], [255, 255, 255, 255]);
                        update_position(b_sprites[i], [sx, sy]);
                        b_active[i] = true;
                        b_type[i] = 0;
                        b_vx[i] = bullet_speed * math_cos(a);
                        b_vy[i] = bullet_speed * math_sin(a);
                        b_life[i] = 0;
                        fired = fired + 1;
                    }
                }
                if (has_backshot) {
                    let done = false;
                    for (let i = 0; i < bullet_pool_size && !done; i = i + 1) {
                        if (!b_active[i]) {
                            let ba = base + math_PI;
                            let bs = bullet_radius / 4;
                            let sx2 = width / 2 + spawn_dist * math_cos(ba);
                            let sy2 = height / 2 + spawn_dist * math_sin(ba);
                            update_scale(b_sprites[i], [bs, bs]);
                            update_rotation(b_sprites[i], ba + math_PI / 2);
                            update_color(b_sprites[i], [255, 255, 255, 255]);
                            update_position(b_sprites[i], [sx2, sy2]);
                            b_active[i] = true;
                            b_type[i] = 0;
                            b_vx[i] = bullet_speed * math_cos(ba);
                            b_vy[i] = bullet_speed * math_sin(ba);
                            b_life[i] = 0;
                            done = true;
                        }
                    }
                }
            };
        }

        if (message === "update_all") {
            return () => {
                let ec3 = array_length(enermys);
                for (let i = 0; i < bullet_pool_size; i = i + 1) {
                    if (b_active[i]) {
                        b_life[i] = b_life[i] + 1;
                        let sf = 1 - b_life[i] / bullet_life;
                        let cvx = b_vx[i] * sf;
                        let cvy = b_vy[i] * sf;
                        let pos = query_position(b_sprites[i]);
                        let nx = pos[0] + cvx;
                        let ny = pos[1] + cvy;
                        update_position(b_sprites[i], [nx, ny]);
                        let mpx = player_map_px + (nx - width / 2);
                        let mpy = player_map_py + (ny - height / 2);
                        let tx = math_floor(mpx / tile_pixel);
                        let ty = math_floor(mpy / tile_pixel);
                        let hit = false;
                        for (let j = 0; j < ec3 && !hit; j = j + 1) {
                            if (enermys[j]('is_alive')()) {
                                let ep = query_position(enermys[j]('get_sprite')());
                                let edx = nx - ep[0];
                                let edy = ny - ep[1];
                                if (math_abs(edx) < 24 && math_abs(edy) < 24 &&
                                    edx * edx + edy * edy < 576) {
                                    
                                    hit = true;
                                    
                                    // 伤害判别与电击手枪特化效果
                                    if (b_type[i] === 2) { 
                                        // 【电击枪效果】：对大范围内的所有怪物静止 180 帧 (3 秒)，并将生命值削至 1
                                        let stun_radius = 250;
                                        for (let k = 0; k < ec3; k = k + 1) {
                                            if (enermys[k]('is_alive')()) {
                                                let target_p = query_position(enermys[k]('get_sprite')());
                                                let tdx = ep[0] - target_p[0];
                                                let tdy = ep[1] - target_p[1];
                                                if (tdx * tdx + tdy * tdy < stun_radius * stun_radius) {
                                                    enermys[k]('stun')(180);
                                                    enermys[k]('set_hp')(1);
                                                }
                                            }
                                        }
                                        enermys[j]('damage')(1);
                                    } else if (b_type[i] === 1) {
                                        enermys[j]('damage')(999); // 左轮秒杀
                                    } else {
                                        enermys[j]('damage')(1);
                                    }
                                    
                                    if (has_knockback) {
                                        let bv = enermys[j]('get_behavior')();
                                        let kstr = 15;
                                        if (bv === 3 || bv === 4) { kstr = 5; }
                                        let pdx = ep[0] - width / 2;
                                        let pdy = ep[1] - height / 2;
                                        let pdist = math_sqrt(pdx * pdx + pdy * pdy);
                                        if (pdist > 0) {
                                            enermys[j]('push')(pdx / pdist * kstr, pdy / pdist * kstr);
                                        }
                                    }
                                }
                            }
                        }
                        if (sf <= 0 || is_wall(tile_map, tx, ty) || hit) {
                            b_active[i] = false;
                            update_scale(b_sprites[i], [0, 0]);
                            update_position(b_sprites[i], [-9999, -9999]);
                        }
                    }
                }
            };
        }
        return 0;
    };
}

// 移动边界阻挡逻辑
function is_wall(tile_map, x, y) {
    if (x < 0 || x >= array_length(tile_map[0]) ||
        y < 0 || y >= array_length(tile_map)) {
        return true;
    }
    let t = tile_map[y][x];
    return t === 'wallTop' ||
           t === 'wallCenter' ||
           t === 'wallLeftBottom' || t === 'bottomWall' ||
           t === 'bottomRightWall' || t === 'rightWall' ||
           t === 'LeftWall' ||
           t === 'topLeftWall' || t === 'topWall' ||
           t === 'topRightWall';
}

function can_move_x(tile_map, px, py, dx) {
    if (dx > 0) {
        let check_x = math_floor((px + dx + tile_pixel - 1) / tile_pixel);
        let top_y   = math_floor(py / tile_pixel);
        let bot_y   = math_floor((py + tile_pixel - 1) / tile_pixel);
        return !(is_wall(tile_map, check_x, top_y) ||
                 is_wall(tile_map, check_x, bot_y));
    } else if (dx < 0) {
        let check_x = math_floor((px + dx) / tile_pixel);
        let top_y   = math_floor(py / tile_pixel);
        let bot_y   = math_floor((py + tile_pixel - 1) / tile_pixel);
        return !(is_wall(tile_map, check_x, top_y) ||
                 is_wall(tile_map, check_x, bot_y));
    }
    return true;
}

function can_move_y(tile_map, px, py, dy) {
    if (dy > 0) {
        let left_x  = math_floor(px / tile_pixel);
        let right_x = math_floor((px + tile_pixel - 1) / tile_pixel);
        let check_y = math_floor((py + dy + tile_pixel - 1) / tile_pixel);
        return !(is_wall(tile_map, left_x, check_y) ||
                 is_wall(tile_map, right_x, check_y));
    } else if (dy < 0) {
        let left_x  = math_floor(px / tile_pixel);
        let right_x = math_floor((px + tile_pixel - 1) / tile_pixel);
        let check_y = math_floor((py + dy) / tile_pixel);
        return !(is_wall(tile_map, left_x, check_y) ||
                 is_wall(tile_map, right_x, check_y));
    }
    return true;
}

// 输入监听
function input_listener() { 
    let dx = 0;
    let dy = 0;
    
    if (input_key_down('w')) { dy = dy - player_speed; }
    if (input_key_down('s')) { dy = dy + player_speed; }
    if (input_key_down('a')) { 
        dx = dx - player_speed; 
        player()('determine_direction')(player_instance, dx);
    }
    if (input_key_down('d')) {
        dx = dx + player_speed;
        player()('determine_direction')(player_instance, dx);
    }
    
    if (dx !== 0 && dy !== 0) {
        let diag = player_speed / math_sqrt(2);
        dx = dx > 0 ? diag : -diag;
        dy = dy > 0 ? diag : -diag;
    }
    
    player_moving = (dx !== 0 || dy !== 0);
    
    let ec = array_length(enermys);
    let tc = array_length(tile_sprites);
    
    if (dx !== 0 && can_move_x(tile_map, player_map_px, player_map_py, dx)) {
        let was_hit = false;
        for (let i = 0; i < ec && !was_hit; i = i + 1) {
            if (enermys[i]('is_alive')() &&
                gameobjects_overlap(player_instance, enermys[i]('get_sprite')())) {
                was_hit = true;
            }
        }
        player_map_px = player_map_px + dx;
        for (let i = 0; i < tc; i = i + 1) {
            let pos = query_position(tile_sprites[i]);
            update_position(tile_sprites[i], [pos[0] - dx, pos[1]]);
        }
        for (let i = 0; i < ec; i = i + 1) {
            enermys[i]('update_map_offset')(dx, 0);
        }
        enemy_bullets()('update_map_offset')(dx, 0);
        game_rewards()('update_map_offset')(dx, 0);
        this_bomb('update_map_offset')(dx, 0);
        
        let now_hit = false;
        for (let i = 0; i < ec && !now_hit; i = i + 1) {
            if (enermys[i]('is_alive')() &&
                gameobjects_overlap(player_instance, enermys[i]('get_sprite')())) {
                now_hit = true;
            }
        }
        if (!was_hit && now_hit) {
            player_map_px = player_map_px - dx;
            for (let i = 0; i < tc; i = i + 1) {
                let pos = query_position(tile_sprites[i]);
                update_position(tile_sprites[i], [pos[0] + dx, pos[1]]);
            }
            for (let i = 0; i < ec; i = i + 1) {
                enermys[i]('update_map_offset')(-dx, 0);
            }
            enemy_bullets()('update_map_offset')(-dx, 0);
            game_rewards()('update_map_offset')(-dx, 0);
            this_bomb('update_map_offset')(-dx, 0);
        }
    }
    
    if (dy !== 0 && can_move_y(tile_map, player_map_px, player_map_py, dy)) {
        let was_hit = false;
        for (let i = 0; i < ec && !was_hit; i = i + 1) {
            if (enermys[i]('is_alive')() &&
                gameobjects_overlap(player_instance, enermys[i]('get_sprite')())) {
                was_hit = true;
            }
        }
        player_map_py = player_map_py + dy;
        for (let i = 0; i < tc; i = i + 1) {
            let pos = query_position(tile_sprites[i]);
            update_position(tile_sprites[i], [pos[0], pos[1] - dy]);
        }
        for (let i = 0; i < ec; i = i + 1) {
            enermys[i]('update_map_offset')(0, dy);
        }
        enemy_bullets()('update_map_offset')(0, dy);
        game_rewards()('update_map_offset')(0, dy);
        this_bomb('update_map_offset')(0, dy);
        
        let now_hit = false;
        for (let i = 0; i < ec && !now_hit; i = i + 1) {
            if (enermys[i]('is_alive')() &&
                gameobjects_overlap(player_instance, enermys[i]('get_sprite')())) {
                now_hit = true;
            }
        }
        if (!was_hit && now_hit) {
            player_map_py = player_map_py - dy;
            for (let i = 0; i < tc; i = i + 1) {
                let pos = query_position(tile_sprites[i]);
                update_position(tile_sprites[i], [pos[0], pos[1] + dy]);
            }
            for (let i = 0; i < ec; i = i + 1) {
                enermys[i]('update_map_offset')(0, -dy);

update_loop((game_state) => {
    if (!game_started) {
        update_to_top(start_bg);
        update_to_top(start_title);
        update_to_top(start_go_sprite);

        if (input_left_mouse_down()) {
            const mp = query_pointer_position();
            const btnPos = query_position(start_go_sprite);
            const btnW = 200;
            const btnH = 100;
            if (mp[0] > btnPos[0] - btnW && mp[0] < btnPos[0] + btnW &&
                mp[1] > btnPos[1] - btnH && mp[1] < btnPos[1] + btnH) {
                game_started = true;
                update_position(start_bg, [-9999, -9999]);
                update_scale(start_bg, [0, 0]);
                update_position(start_title, [-9999, -9999]);
                update_scale(start_title, [0, 0]);
                update_position(start_go_sprite, [-9999, -9999]);
                update_scale(start_go_sprite, [0, 0]);
            }
        }
    } else {
        if (player_hp <= 0) {
            gameover_active = true;
        }
        if (gameover_active) {
            update_position(gameover_card, [width / 2, height / 2]);
            update_scale(gameover_card, [4, 1.2]);
            update_color(gameover_card, [30, 30, 30, 255]);
            update_to_top(gameover_card);
            update_text(gameover_text, 'GAME OVER');
            update_position(gameover_text, [width / 2, height / 2]);
            update_scale(gameover_text, [1, 1]);
            update_to_top(gameover_text);
        }
        if (!levelup_active && !gameover_active) {
            const barX = 25;
            const barY = 25;
            
            fireLightCover()('update_fire_light')(fireLighterCover);
            
            update_position(hpBarBackground, [barX, barY]);
            update_position(hpBarFill, [barX, barY]);
            update_to_top(hpBarBackground);
            update_to_top(hpBarFill);
            let hpRatio = player_hp / player_max_hp;
            update_scale(hpBarFill, [hpRatio, 1]);
            let xpNeed = 3 + 2 * player_level;
            let xpRatio = player_kills / xpNeed;
            update_position(xpBarBackground, [width / 2, 4]);
            update_position(xpBarFill, [width / 2, 4]);
            update_to_top(xpBarBackground);
            update_to_top(xpBarFill);
            update_scale(xpBarFill, [xpRatio, 1]);
            if (player_hurt_cd > 0) { player_hurt_cd = player_hurt_cd - 1; }
            player()('update_color')(player_instance);

            if (player_moving) {
                player_anim_timer = player_anim_timer + 1;
                if (player_anim_timer >= 25) {
                    player_anim_timer = 0;
                    if (query_scale(player_instance)[0] !== 0) {
                        update_scale(player_instance, [0, 0]);
                        update_scale(player_frame2, [player_dir_sign * player_scale, player_scale]);
                    } else {
                        update_scale(player_instance, [player_dir_sign * player_scale, player_scale]);
                        update_scale(player_frame2, [0, 0]);
                    }
                }
            } else {
                update_scale(player_instance, [player_dir_sign * player_scale, player_scale]);
                update_scale(player_frame2, [0, 0]);
            }
            player()('update_color')(player_frame2);

            input_listener();

            spawner_instance('update')();
            game_rewards()('update')();
            
            gun()('update_gun')(gun_instance);

            // 获取鼠标方向并实时更新左轮/电击枪射线位置
            let mp_aim = query_pointer_position();
            let aim_dx = mp_aim[0] - width / 2;
            let aim_dy = mp_aim[1] - height / 2;
            let aim_angle = math_atan2(aim_dy, aim_dx);
            this_revolver('update_aim_line')(width / 2, height / 2, aim_angle);
            this_taser('update_aim_line')(width / 2, height / 2, aim_angle);
        
            if (fire_cooldown > 0) {
                fire_cooldown = fire_cooldown - 1;
            }

            // 【武器开火与模式判定】
            if (input_left_mouse_down() && fire_cooldown <= 0) {
                fireLightCoverTimer = 2;
                play_audio(shoot_sound);
                let mp = query_pointer_position();

                if (is_bomb_mode && bomb_ammo > 0) {
                    let dx2 = mp[0] - width / 2;
                    let dy2 = mp[1] - height / 2;
                    let angle = math_atan2(dy2, dx2);
                    this_bomb('fire')(width / 2, height / 2, angle);
                    bomb_ammo = bomb_ammo - 1;
                    if (bomb_ammo <= 0) { 
                        is_bomb_mode = false; 
                        bomb_ammo = 0; 
                    }
                    fire_cooldown = bullet_fire_rate;
                } else if (is_gatling_mode && gatling_ammo > 0) {
                    let dx2 = mp[0] - width / 2;
                    let dy2 = mp[1] - height / 2;
                    let angle = math_atan2(dy2, dx2);
                    
                    this_gatling('fire')(width / 2, height / 2, angle);
                    gatling_ammo = gatling_ammo - 1;
                    
                    if (gatling_ammo <= 0) { 
                        is_gatling_mode = false; 
                        gatling_ammo = 0; 
                    }
                    fire_cooldown = this_gatling('get_fire_rate')();
                } else if (is_revolver_mode && revolver_ammo > 0) {
                    this_revolver('fire')(width / 2, height / 2, aim_angle);
                    
                    revolver_ammo = revolver_ammo - 1;
                    if (revolver_ammo <= 0) {
                        is_revolver_mode = false;
                        revolver_ammo = 0;
                        this_revolver('update_aim_line')(0, 0, 0); // 隐藏射线
                    }
                    
                    fire_cooldown = this_revolver('get_fire_rate')();
                } else if (is_taser_mode && taser_ammo > 0) {
                    this_taser('fire')(width / 2, height / 2, aim_angle);
                    
                    taser_ammo = taser_ammo - 1;
                    if (taser_ammo <= 0) {
                        is_taser_mode = false;
                        taser_ammo = 0;
                        this_taser('update_aim_line')(0, 0, 0); // 隐藏射线
                    }
                    
                    fire_cooldown = this_taser('get_fire_rate')();
                } else {
                    bullets()('fire')(mp);
                    fire_cooldown = bullet_fire_rate;
                }
            }

            bullets()('update_all')();
            this_bomb('update')();
            
            particles()('update_all')();
            enemy_bullets()('update_all')();
            
            let ec2 = array_length(enermys);
            for (let i = 0; i < ec2; i = i + 1) {
                enermys[i]('chase_player')();
            }
            for (let i = 0; i < ec2; i = i + 1) {
                for (let j = i + 1; j < ec2; j = j + 1) {
                    if (enermys[i]('is_alive')() && enermys[j]('is_alive')()) {
                        let p1 = query_position(enermys[i]('get_sprite')());
                        let p2 = query_position(enermys[j]('get_sprite')());
                        let edx = p1[0] - p2[0];
                        let edy = p1[1] - p2[1];
                        if (math_abs(edx) < 40 && math_abs(edy) < 40) {
                            let edsq = edx * edx + edy * edy;
                            if (edsq < 1600 && edsq > 0) {
                                let ed = math_sqrt(edsq);
                                let overlap = (40 - ed) / 2;
                                let px = (edx / ed) * overlap;
                                let py = (edy / ed) * overlap;
                                enermys[i]('push')(px, py);
                                enermys[j]('push')(-px, -py);
                            }
                        }
                    }
                }
            }
            
            for (let i = 0; i < ec2; i = i + 1) {
                if (enermys[i]('is_alive')()) {
                    let p_enemy = query_position(enermys[i]('get_sprite')());
                    let edx = p_enemy[0] - width / 2;
                    let edy = p_enemy[1] - height / 2;
                    let edsq = edx * edx + edy * edy;
                    if (edsq < 900 && edsq > 0) {
                        let ed = math_sqrt(edsq);
                        let overlap = 30 - ed;
                        let px = (edx / ed) * overlap;
                        let py = (edy / ed) * overlap;
                        enermys[i]('push')(px, py);
                        if (player_hurt_cd <= 0) {
                            player()('damage')(1);
                            player_hurt_cd = player_hurt_interval;
                        }
                    }
                }
            }
        }
        
        if (levelup_click_lock && !input_left_mouse_down()) {
            levelup_click_lock = false;
        }
        if (levelup_active && !levelup_click_lock && input_left_mouse_down()) {
            let mp = query_pointer_position();
            for (let i = 0; i < 3; i = i + 1) {
                let cx = width / 2 + (i - 1) * 180;
                let cy = height / 2;
                if (mp[0] > cx - 80 && mp[0] < cx + 80 &&
                    mp[1] > cy - 30 && mp[1] < cy + 30) {
                    apply_upgrade(levelup_options[i]);
                    hide_levelup();
                }
            }
        }
    }

    if (game_state[0] === undefined) {
        game_state[0] = 0;
    }
    game_state[0] = game_state[0] + 1;
});
